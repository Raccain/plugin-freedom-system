# RadioMusic - CMake WebView Configuration Snippet
# Add this to your existing CMakeLists.txt

# Enable WebView support in JUCE
juce_add_plugin(RadioMusic
    COMPANY_NAME "YourCompany"
    PLUGIN_MANUFACTURER_CODE Manu
    PLUGIN_CODE Rmsc
    FORMATS AU VST3 Standalone
    PRODUCT_NAME "RadioMusic"

    # WebView requirement
    NEEDS_WEB_BROWSER TRUE

    # Other settings
    IS_SYNTH FALSE
    NEEDS_MIDI_INPUT TRUE
    NEEDS_MIDI_OUTPUT FALSE
    IS_MIDI_EFFECT FALSE
    EDITOR_WANTS_KEYBOARD_FOCUS FALSE
    COPY_PLUGIN_AFTER_BUILD TRUE
)

# Add binary resources for HTML/CSS/JS
juce_add_binary_data(RadioMusicAssets
    SOURCES
        Source/ui/ui.html
)

# Link binary data to plugin
target_link_libraries(RadioMusic
    PRIVATE
        RadioMusicAssets
        juce::juce_audio_utils
        juce::juce_gui_extra  # Required for WebBrowserComponent
    PUBLIC
        juce::juce_recommended_config_flags
        juce::juce_recommended_lto_flags
        juce::juce_recommended_warning_flags
)

# Preprocessor definitions
target_compile_definitions(RadioMusic
    PUBLIC
        JUCE_WEB_BROWSER=1
        JUCE_USE_CURL=0
        JUCE_VST3_CAN_REPLACE_VST2=0
        JUCE_DISPLAY_SPLASH_SCREEN=0
)
